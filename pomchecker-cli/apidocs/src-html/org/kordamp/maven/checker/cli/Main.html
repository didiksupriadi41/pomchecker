018package org.kordamp.maven.checker.cli;
019
020import picocli.CommandLine;
021
022import java.io.PrintWriter;
023
024/**
025 * @author Andres Almiray
026 * @since 1.1.0
027 */
028@CommandLine.Command(name = "pomchecker",
029    description = "pomchecker",
030    mixinStandardHelpOptions = true,
031    versionProvider = Versions.class,
032    subcommands = {CheckBom.class, CheckMavenCentral.class})
033public class Main implements Runnable {
034    PrintWriter out;
035    PrintWriter err;
036
037    @CommandLine.Spec
038    CommandLine.Model.CommandSpec spec;
039
040    public void run() {
041        Banner.display(out);
042
043        spec.commandLine().usage(out);
044    }
045
046    public static void main(String[] args) {
047        System.exit(run(args));
048    }
049
050    public static int run(String... args) {
051        Main cmd = new Main();
052        CommandLine commandLine = new CommandLine(cmd);
053        cmd.out = commandLine.getOut();
054        cmd.err = commandLine.getErr();
055        return commandLine.execute(args);
056    }
057
058    public static int run(PrintWriter out, PrintWriter err, String... args) {
059        Main cmd = new Main();
060        cmd.out = out;
061        cmd.err = err;
062        return new CommandLine(cmd).execute(args);
063    }
064}
