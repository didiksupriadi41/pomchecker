018package org.kordamp.maven.enforcer.checker;
019
020import org.apache.maven.enforcer.rule.api.EnforcerRule;
021import org.apache.maven.enforcer.rule.api.EnforcerRuleException;
022import org.apache.maven.enforcer.rule.api.EnforcerRuleHelper;
023import org.apache.maven.project.MavenProject;
024import org.codehaus.plexus.component.configurator.expression.ExpressionEvaluationException;
025import org.kordamp.maven.checker.MavenCentralChecker;
026import org.kordamp.maven.checker.MavenLoggerAdapter;
027import org.kordamp.maven.checker.PomCheckException;
028
029import javax.annotation.Nonnull;
030import javax.annotation.Nullable;
031
032/**
033 * Checks if a POM complies with the rules for uploading to Maven Central.
034 *
035 * @author Andres Almiray
036 * @since 1.0.0
037 */
038public class CheckMavenCentral implements EnforcerRule {
039    private boolean release = true;
040
041    private boolean strict = true;
042
043    @Override
044    public void execute(@Nonnull EnforcerRuleHelper helper) throws EnforcerRuleException {
045        try {
046            MavenProject project = (MavenProject) helper.evaluate("${project}");
047            MavenCentralChecker.check(new MavenLoggerAdapter(helper.getLog()), project, release, strict);
048        } catch (ExpressionEvaluationException e) {
049            throw new EnforcerRuleException("Unable to lookup an expression " + e.getLocalizedMessage(), e);
050        } catch (PomCheckException e) {
051            throw new EnforcerRuleException(e.getMessage());
052        }
053    }
054
055    @Override
056    public boolean isCacheable() {
057        return false;
058    }
059
060    @Override
061    public boolean isResultValid(@Nonnull EnforcerRule enforcerRule) {
062        return false;
063    }
064
065    @Nullable
066    @Override
067    public String getCacheId() {
068        return "";
069    }
070}
