018package org.kordamp.maven.checker.toolprovider;
019
020import org.kordamp.jipsy.ServiceProviderFor;
021import org.kordamp.maven.checker.cli.Main;
022
023import java.io.PrintWriter;
024import java.util.spi.ToolProvider;
025
026/**
027 * @author Andres Almiray
028 * @since 1.1.0
029 */
030@ServiceProviderFor(ToolProvider.class)
031public class Pomchecker implements ToolProvider {
032    public String name() {
033        return "pomchecker";
034    }
035
036    public int run(PrintWriter out, PrintWriter err, String... args) {
037        return Main.run(out, err, args);
038    }
039
040    public static void main(String[] args) {
041        PrintWriter out = new PrintWriter(System.out);
042        PrintWriter err = new PrintWriter(System.err);
043
044        try {
045            Main.run(out, err, args);
046        } finally {
047            out.flush();
048            err.flush();
049        }
050    }
051}
