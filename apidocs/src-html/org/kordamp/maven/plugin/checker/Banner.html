018package org.kordamp.maven.plugin.checker;
019
020import org.apache.maven.plugin.logging.Log;
021import org.apache.maven.project.MavenProject;
022
023import java.io.File;
024import java.io.FileInputStream;
025import java.io.FileOutputStream;
026import java.io.IOException;
027import java.io.PrintStream;
028import java.text.MessageFormat;
029import java.util.ArrayList;
030import java.util.List;
031import java.util.ResourceBundle;
032import java.util.Scanner;
033
034/**
035 * @author Andres Almiray
036 * @since 1.0.0
037 */
038public final class Banner {
039    private static final Banner b = new Banner();
040    private final ResourceBundle bundle = ResourceBundle.getBundle(Banner.class.getName());
041    private final String productVersion = bundle.getString("product.version");
042    private final String productId = bundle.getString("product.id");
043    private final String productName = bundle.getString("product.name");
044    private final String banner = MessageFormat.format(bundle.getString("product.banner"), productName, productVersion);
045    private final List<String> visited = new ArrayList<>();
046
047    private Banner() {
048        // nooop
049    }
050
051    public static void display(MavenProject project, Log log) {
052        MavenProject element = project;
053        MavenProject root = project;
054        while (true) {
055            if (element.getParent() == null || element.getParent() == element) {
056                break;
057            }
058            root = element.getParent();
059        }
060        if (b.visited.contains(root.getName())) {
061            return;
062        }
063
064        b.visited.add(root.getName());
065
066        try {
067            File parent = new File(System.getProperty("user.home"), "/.m2/caches");
068            File markerFile = b.getMarkerFile(parent);
069            if (!markerFile.exists()) {
070                System.out.println(b.banner);
071                markerFile.getParentFile().mkdirs();
072                PrintStream out = new PrintStream(new FileOutputStream(markerFile));
073                out.println("1");
074                out.close();
075                writeQuietly(markerFile, "1");
076            } else {
077                try {
078                    int count = Integer.parseInt(readQuietly(markerFile));
079                    if (count < 3) {
080                        System.out.println(b.banner);
081                    }
082                    writeQuietly(markerFile, (count + 1) + "");
083                } catch (NumberFormatException e) {
084                    writeQuietly(markerFile, "1");
085                    System.out.println(b.banner);
086                }
087            }
088        } catch (IOException ignored) {
089            // noop
090        }
091    }
092
093    private static void writeQuietly(File file, String text) {
094        try {
095            PrintStream out = new PrintStream(new FileOutputStream(file));
096            out.println(text);
097            out.close();
098        } catch (IOException ignored) {
099            // ignored
100        }
101    }
102
103    private static String readQuietly(File file) {
104        try {
105            Scanner in = new Scanner(new FileInputStream(file));
106            return in.next();
107        } catch (Exception ignored) {
108            return "";
109        }
110    }
111
112    private File getMarkerFile(File parent) {
113        return new File(parent,
114            "kordamp" +
115                File.separator +
116                productId +
117                File.separator +
118                productVersion +
119                File.separator +
120                "marker.txt");
121    }
122}
