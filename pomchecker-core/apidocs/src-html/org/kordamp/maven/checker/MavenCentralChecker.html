028import java.io.File;

030import java.util.ArrayList;
031import java.util.List;

033import static org.apache.commons.lang3.StringUtils.isBlank;
034import static org.apache.commons.lang3.StringUtils.isNotBlank;

036/**
037 * Checks if a POM complies with the rules for uploading to Maven Central.
038 *
039 * The following blocks are required:
040 * <ul>
041 *   <li>&lt;groupId&gt;</li>
042 *   <li>&lt;artifactId&gt;</li>
043 *   <li>&lt;version&gt;</li>
044 *   <li>&lt;name&gt;</li>
045 *   <li>&lt;description&gt;</li>
046 *   <li>&lt;url&gt;</li>
047 *   <li>&lt;licenses&gt;</li>
048 *   <li>&lt;scm&gt;</li>
049 * </ul>
050 *
051 * All previous blocks may be supplied by a parent POM with the exception of &lt;artifactId&gt;.
052 *
053 * The following blocks are forbidden if {@code strict = true}
054 * <ul>
055 *   <li>&lt;repositories&gt;</li>
056 *   <li>&lt;pluginRepositories&gt;</li>
057 * </ul>
058 *
059 * @see <a href="http://maven.apache.org/repository/guide-central-repository-upload.html">http://maven.apache.org/repository/guide-central-repository-upload.html</a>
060 * @see <a href="https://central.sonatype.org/pages/requirements.html">https://central.sonatype.org/pages/requirements.html</a>
061 *
062 * @author Andres Almiray
063 * @since 1.0.0
064 */
065public class MavenCentralChecker {
066    /**
067     * Checks the resolved model of the given MaveProject for compliance.
068     *
069     * @param log     the logger to use.
070     * @param project the project to be checked.
071     * @param release if {@code true} checks if version is not -SNAPSHOT.
072     * @param strict  if {@code true} checks that &lt;repositories&gt; and &lt;pluginRepositories&gt; are not present
073     * @throws PomCheckException if the POM is invalid
074     */
075    public static void check(Log log, MavenProject project, boolean release, boolean strict) throws PomCheckException {
076        Model fullModel = project.getModel();
077        Model originalModel = project.getOriginalModel();
078
079        List<String> errors = new ArrayList<>();
080
081        // sanity checks. redundant?
082        if (isBlank(fullModel.getGroupId())) {
083            errors.add("<groupId> cannot not be blank.");
084        }
085        if (isBlank(fullModel.getArtifactId())) {
086            errors.add("<artifactId> cannot not be blank.");
087        }
088        if (isBlank(fullModel.getVersion())) {
089            errors.add("<version> cannot not be blank.");
090        }
091
092        if (isBlank(fullModel.getName())) {
093            String parentName = resolveParentName(project.getFile().getParentFile(), fullModel);
094            if (isBlank(parentName)) {
095                errors.add("<name> cannot not be blank.");
096            } else {
097                log.warn("<name> is not defined in POM. Will use value from parent: " +
098                    System.lineSeparator() + "\t" + parentName);
099            }
100        }
101
102        if (isBlank(fullModel.getDescription())) {
103            errors.add("<description> cannot not be blank.");
104        }
105        if (isBlank(originalModel.getDescription())) {
106            log.warn("<description> is not defined in POM. Will use value from parent: " +
107                System.lineSeparator() + "\t" + fullModel.getDescription());
108        }
109
110        if (isBlank(fullModel.getUrl())) {
111            errors.add("<url> cannot not be blank.");
112        }
113        if (isBlank(originalModel.getUrl())) {
114            log.warn("<url> is not defined in POM. Will use computed value from parent: " +
115                System.lineSeparator() + "\t" + fullModel.getUrl());
116        }
117
118        if (release && fullModel.getVersion().endsWith("-SNAPSHOT")) {
119            errors.add("<version> cannot not be -SNAPSHOT.");
120        }
121
122        if (fullModel.getLicenses() != null) {
123            if (!fullModel.getLicenses().isEmpty()) {
124                // verify that all licenses have <name> & <url>
125                for (int i = 0; i < fullModel.getLicenses().size(); i++) {
126                    License license = fullModel.getLicenses().get(i);
127                    if (isBlank(license.getName()) && isBlank(license.getUrl())) {
128                        errors.add("License " + i + " must define <name> and <url>.");
129                    }
130                }
131            } else {
132                errors.add("<licenses> block is required but was left empty.");
133            }
134        } else {
135            errors.add("<licenses> block is required but was left undefined.");
136        }
137
138        if (fullModel.getDevelopers() != null) {
139            if (!fullModel.getDevelopers().isEmpty()) {
140                // verify that all developers have at least one of [id, name, organization, organizationUrl]
141                for (int i = 0; i < fullModel.getDevelopers().size(); i++) {
142                    Developer developer = fullModel.getDevelopers().get(i);
143                    if (isBlank(developer.getId()) &&
144                        isBlank(developer.getName()) &&
145                        isBlank(developer.getOrganization()) &&
146                        isBlank(developer.getOrganizationUrl())) {
147                        errors.add("Developer " + i + " must define at least one of <id>, <name>, <organization>, <organizationUrl>.");
148                    }
149                }
150            } else {
151                errors.add("<developers> block is required but was left empty.");
152            }
153        } else {
154            errors.add("<developers> block is required but was left undefined.");
155        }
156
157        if (fullModel.getScm() == null) {
158            errors.add("The <scm> block is required.");
159        }
160
161        if (null != originalModel.getRepositories() && !originalModel.getRepositories().isEmpty()) {
162            if (strict) {
163                errors.add("The <repositories> block should not be present.");
164            } else {
165                log.warn("The <repositories> block should not be present.");
166            }
167        }
168        if (null != originalModel.getPluginRepositories() && !originalModel.getPluginRepositories().isEmpty()) {
169            if (strict) {
170                errors.add("The <pluginRepositories> block should not be present.");
171            } else {
172                log.warn("The <pluginRepositories> block should not be present.");
173            }
174        }
175
176        if (!errors.isEmpty()) {
177            StringBuilder b = new StringBuilder("\nThe POM file\n")
178                .append(project.getFile().getAbsolutePath())
179                .append("\ncannot be uploaded to Maven Central due to the following reasons:\n");
180            for (String s : errors) {
181                b.append(" * ").append(s).append("\n");
182            }
183
184            throw new PomCheckException(b.toString());
185        }
186    }
187
188    private static String resolveParentName(File directory, Model fullModel) {
189        Parent parent = fullModel.getParent();
190
191        while (parent != null) {
192            if (isNotBlank(parent.getRelativePath())) {
193                File pomFile = new File(directory, parent.getRelativePath());
194                if (!pomFile.getName().endsWith("pom.xml")) {
195                    pomFile = new File(pomFile, "pom.xml");
196                }
197
198                if (pomFile.exists()) {
199                    MavenProject parentProject = readProject(pomFile);
200                    Model parentModel = parentProject.getModel();
201                    if (isNotBlank(parentModel.getName())) {
202                        return parentModel.getName();
203                    } else {
204                        directory = pomFile.getParentFile();
205                        parent = parentModel.getParent();
206                    }
207                } else {
208                    // parent should be available from a repository
209                    // TODO: resolve parent
210                    return null;
211                }
212            } else {
213                // check 1 level up
214                File pomFile = new File(directory, "../pom.xml");
215                if (pomFile.exists()) {
216                    MavenProject parentProject = readProject(pomFile);
217                    Model parentModel = parentProject.getModel();
218                    if (isNotBlank(parentModel.getName())) {
219                        return parentModel.getName();
220                    } else {
221                        directory = pomFile.getParentFile();
222                        parent = parentModel.getParent();
223                    }
224                } else {
225                    // parent should be available form a repository
226                    // TODO: resolve parent
227                    return null;
228                }
229            }
230        }
231
232        return null;
233    }
234
235    private static MavenProject readProject(File pom) {
236        try {
237            FileReader reader = new FileReader(pom);
238            MavenXpp3Reader mavenReader = new MavenXpp3Reader();
239            return new MavenProject(mavenReader.read(reader));
240        } catch (Exception e) {
241            throw new IllegalArgumentException(e);
242        }
243    }
